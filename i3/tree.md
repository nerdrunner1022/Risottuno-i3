~/.config/i3/
├── config            # Main config that includes all modules
├── keybindings       # All your keyboard shortcuts
├── workspaces        # Workspace definitions and layouts
├── appearance        # Gaps, borders, fonts, colors, bar config
├── autostart         # Apps or scripts to launch on startup
├── rules             # Floating/window rules for specific apps
└── scripts/          # Custom scripts called from keybindings or bar
